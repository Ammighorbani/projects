5- (create mysql table)
CREATE TABLE users_account (
    id INT AUTO_INCREMENT PRIMARY KEY,
    first_name TEXT CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci NOT NULL,
    last_name TEXT CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci NOT NULL,
    email VARCHAR(255) CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci NOT NULL UNIQUE,
    username VARCHAR(50) CHARACTER SET utf8mb4 COLLATE utf8mb4_bin NOT NULL UNIQUE,
    password TEXT NOT NULL,
    age SMALLINT NOT NULL CHECK (age BETWEEN 0 AND 9999),
    phone_number BIGINT NOT NULL,
    country VARCHAR(50) CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci NOT NULL,
    city VARCHAR(50) CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci,
    province VARCHAR(50) CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci,
    home_address TEXT CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci,
    postalcode BIGINT
);
